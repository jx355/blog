class Solution {
        public String longestPalindrome(String s) {
            int len = s.length();
            if(len<2)
                return s;
            char[] ch = s.toCharArray();
            boolean[][] bool = new boolean[len][len];
            //鍒濆嬪寲bool  姣忎釜鍗曚釜瀛楃﹀潎涓哄洖鏂
            for(int i = 0;i<len;i++){
                bool[i][i] = true;
            }
            int maxlen = 1;
            int begin = 0;

            for(int L = 2;L < len;L++){
                for(int i=0;i<len;i++){
                    int j = L+i-1;
                    if(j>len)
                        break;
                    if(ch[i]!=ch[j]){
                        bool[i][j] = false;
                    }else{
                        if(j-i<3){
                            bool[i][j] = true;
                        }else{
                            bool[i][j] = bool[i+1][j-1];
                        }
                    }
                    if(bool[i][j] && maxlen < j-i+1){
                        begin = i;
                        maxlen = L;
                    }
                }
            }
            return s.substring(begin,begin+maxlen);
        }
    }

    public static int projectionArea(int[][] grid) {
        int lenx = grid.length;
        if(lenx == 0)
            return 0;
        //缁熻￠潰绉
        int x=0,y=0,z=0;
        //閫氳繃鏁扮粍鏉ュ瓨鍌ㄦ瘡琛屾瘡鍒楃殑鏈澶у  鍦ㄨ＄畻缁撴潫鍚庨渶瑕侀噸鏂伴亶鍘嗚幏鍙栨诲拰
        int[] countX = new int[lenx];
        int[] countY = new int[lenx];
        //閬嶅巻grid 鑾峰彇z 鍙奵ountX countY
        for(int i = 0; i<lenx; i++){
            countX[i] = grid[i][0];
            countY[i] = grid[0][i];
            for(int j = 0; j<lenx; j++){
                if(grid[i][j]!=0){
                    z++;
                }
                if(countX[i] < grid[i][j]){
                    countX[i] = grid[i][j];
                }
                if(countY[i] < grid[j][i]){
                    countY[i] = grid[j][i];
                }
            }
        }
        int sum = z;
        sum += Arrays.stream(countX).sum();
        sum += Arrays.stream(countY).sum();
        return sum;
    }
}

class Solution {
    public static int projectionArea(int[][] grid) {
        int lenx = grid.length;
        if(lenx == 0)
            return 0;
        int x=0,y=0,z=0;
        for(int i = 0; i<lenx; i++){
            int xlabel=grid[i][0],ylabel = grid[0][i];
            for(int j = 0; j<lenx; j++){
                if(grid[i][j]!=0)
                {
                    z++;
                }
                xlabel = Math.max(xlabel,grid[i][j]);
                ylabel = Math.max(ylabel,grid[j][i]);
            }
            x += xlabel;
            y += ylabel;
        }
        return x+y+z;
    }
}

class Solution {
    int []nums;
    int p = 0;
    public Solution(int[] nums) {
        this.nums = nums;
    }

    public int pick(int target) {
        for(int i = 0;i<nums.length;i++){
            p++;
            p=p%(nums.length);
            if(nums[p] != target){
                continue;
            }
            return p;
        }
        return -1;
    }
}

class Solution {
    Map<Integer, List<Integer>> pos;
    Random random;

    public Solution(int[] nums) {
        pos = new HashMap<Integer, List<Ineger>>();
        random = new Random();
        for (int i = 0; i < nums.length; ++i) {
            pos.putIfAbsent(nums[i], new ArrayList<Integer>());
            pos.get(nums[i]).add(i);
        }
    }

    public int pick(int target) {
        List<Integer> indices = pos.get(target);
        return indices.get(random.nextInt(indices.size()));
    }
}

class Solution {
    int []nums;
    Random random;
    public Solution(int[] nums) {
        this.nums = nums;
        random = new Random();
    }

    public int pick(int target) {
        int ans = 0;
        for(int i =0,cnt = 0; i<nums.length;i++){
            if(nums[i]==target){
                cnt++;
                if(random.nextInt(cnt)==0)
                ans = i;
            }
        }
        return ans;
    }

class Solution {

    ArrayList<Integer> list = new ArrayList<>(); 

    Random random;

    public Solution(ListNode head) {

        while(head != null){

            list.add(head.val);

            head = head.next;

        }

        random = new Random();

    }

    public int getRandom() {
        //姘村樻娊鏍
        int ans = 0;

        for(int i = 0; i < list.size(); i++){

            if(random.nextInt(i+1)==0){

                ans = i;

            }

        }

        return list.get(ans);
        //涔熷彲鐩存帴杩斿洖闅忔満鍊
        return list.get(random.nextInt(list.size()));

    }
